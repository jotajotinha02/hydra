  <script type="module">

    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
    import { getDatabase, ref, push, onValue } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-database.js";

      apiKey: "AIzaSyB3-GZidpHoYxdrUyQN5cZyKm85bP6kzjw",
      authDomain: "hydra-d3516.firebaseapp.com",
      databaseURL: "https://hydra-d3516-default-rtdb.firebaseio.com",
      projectId: "hydra-d3516",
      storageBucket: "hydra-d3516.firebasestorage.app",
      messagingSenderId: "684571117442",
      appId: "1:684571117442:web:274aefa1dd8b2fdf32b19b",
      measurementId: "G-GHVYJX05ZW"
    };


    const app = initializeApp(firebaseConfig);
    const db = getDatabase(app);
    const vendasRef = ref(db, "vendas"); 
    
    const usuarioInput = document.getElementById("usuario");
    const quantidadeInput = document.getElementById("quantidade");
    const valorInput = document.getElementById("valor");
    const tabelaVendasBody = document.getElementById("tabelaVendas");
    const totalInfoDiv = document.getElementById("totalInfo");
    const filtroDataInput = document.getElementById("filtroData");

    let latestSnapshot = null; 

   
    function renderTableAndTotal(snapshot) {
      latestSnapshot = snapshot; 

      const vendas = [];
      let totalVendas = 0;
      const selectedDate = filtroDataInput.value; 

     
      tabelaVendasBody.innerHTML = "";

     
      snapshot.forEach((childSnapshot) => {
        const venda = childSnapshot.val(); 
        if (!selectedDate || (venda.data && venda.data === selectedDate)) {
          vendas.push(venda);
          totalVendas += venda.valor || 0; 
        }
      });

   
      if (vendas.length === 0) {
       
        const row = tabelaVendasBody.insertRow();
        const cell = row.insertCell();
        cell.colSpan = 4; 
        cell.textContent = selectedDate ? `Nenhuma venda encontrada para a data: ${selectedDate}` : "Nenhuma venda registrada ainda.";
        cell.style.textAlign = 'center';
        cell.style.fontStyle = 'italic';
      } else {
        vendas.forEach(venda => {
          const row = tabelaVendasBody.insertRow();
          row.insertCell().textContent = venda.usuario;
          row.insertCell().textContent = venda.quantidade;
          row.insertCell().textContent = `R$ ${(venda.valor || 0).toFixed(2)}`;
          row.insertCell().textContent = venda.data;
        });
      }

      
      totalInfoDiv.textContent = `Total de vendas filtradas: R$ ${totalVendas.toFixed(2)}`;
    }

   
    onValue(vendasRef, renderTableAndTotal);

 
    function registrarVenda() {
      const usuario = usuarioInput.value.trim(); 
      const quantidade = parseInt(quantidadeInput.value); 
      const valor = parseFloat(valorInput.value); 

    
      if (!usuario || isNaN(quantidade) || isNaN(valor)) {
        alert("Por favor, preencha todos os campos corretamente.");
        return;
      }

     
      const hoje = new Date().toISOString().split("T")[0];

      
      push(vendasRef, {
        usuario: usuario, 
        quantidade: quantidade, 
        valor: valor, 
        data: hoje 
      });

 
      usuarioInput.value = "";
      quantidadeInput.value = "";
      valorInput.value = "";
    }

    function atualizarTabela() {
      if (latestSnapshot) {
        renderTableAndTotal(latestSnapshot);
      }
    }

    window.registrarVenda = registrarVenda;
    window.atualizarTabela = atualizarTabela;
